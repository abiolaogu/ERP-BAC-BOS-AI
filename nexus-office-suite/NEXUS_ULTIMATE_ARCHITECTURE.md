### Sync Engine Architecture

```typescript
interface SyncEngine {
  // Online/offline detection
  detectConnectivity(): Promise<boolean>;

  // Data synchronization
  syncData(entity: string, localData: any): Promise<SyncResult>;
  pullChanges(since: Date): Promise<Change[]>;
  pushChanges(changes: Change[]): Promise<PushResult>;

  // Conflict resolution
  resolveConflicts(conflicts: Conflict[]): Promise<Resolution[]>;

  // Offline queue
  queueOperation(operation: Operation): void;
  processQueue(): Promise<void>;
}
```
